## Install

```bash
npm install --save hanoi-tower-react
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'hanoi-tower-react'
import 'hanoi-tower-react/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
